\section*{C.2 Quick Look-up}
\begin{itemize}
  \item $f_0 \approx \dfrac{1}{2\pi\sqrt{LC}}$ (LC VCO)
  \item $f_{osc} \approx \dfrac{I_{ctrl}}{2 N C_L V_{swing}}$ (ring VCO)
  \item $K_{VCO} \approx \dfrac{g_m}{2 N C_L V_{swing}}$ (linearized)
\end{itemize}
